## Функции, транзакции, pure, view, payable, fallback

```js
// SPDX-License-Identifier: MIT

pragma solidity ^0.8.0;

contract Demo {
    // Области видимости функций:
    // 1) public - означает что данную функцию можно вызывать как из вне контракта (путем отправки специальной транзакции)
    // так и изнутри контракта
    // 2) external - означает что к функции можно обращаться только из вне
    // 3) internal - можно обращаться изнутри контракта - а также те контракты которые наследуют данный контракт
    // 4) private - только изнутри (даже нельзя из потомков)

    // Модификаторы доступа
    // 1) view - означает что данная функция может только читать данные, но не может модифицироватья.
    // это нужно например для того чтобы считать баланс или вернуть какое-то значение.
    //  -- может считать какое-нибудь значение
    //  -- вызывается через специальный колл - и по факту пользователь не должен ничего платить так как по факту 
    //     данные есть на всех узлах. А также надо понимать что по факту тут будет не транзакция а специальный call
    //  выглядит это так
    //  function getBalance public view returns(uint) { ... }
    //  также может быть упрощенная запись:
    //  function getBalanec public voiew returns(uint balance)
    // 2) pure - фактически тоже самое как view но не имеет доступа к внешним полям. то есть тупо нельзя пользоваться 
    // методами класса и проч. Может быть полезна для вычислений по переданным в параметрах данным
    
    // Модификаторы функций вызывающихся через транзакции (за выполнение списываются деньги)
    // 1) отсутствие модификатора
    // например:
    // public setMessage(string memory newMessage) external {}
    // 2) payable - функции принимающие денежные средства
    // пример: function pay() external payable { }
    // данный тип функций автоматически примет средства от клиента и зачислить из на счет контракта
    
    // Fallback функции
    // Если в смартр котракт просто прилетают денежные средства без указания функции 
    // то данные ничего начислено не будет и данные средства просто вернутся на счет отправителя
    // функция декларируется следующим образом
    // receive() external payable { } 
    // данная функция отработает если в смарт контракт прилетели деньги без указания того какую функцию вызвать
    // есть еще одна чистая fallback функция
    // fallback() external payable(не обязательно) {}
    // данная функция вызывается каждый раз когда в контракте была вызвана какая-то левая функция
    // 
    
}
```
